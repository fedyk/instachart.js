# instachart.js

```js
var el = document.getElementById("chart-id");
var chart = instachart.create(el, {
  type: "line" | "bar" | "area",
  data: {
    labels: [1, 2, 3, 4],
    datasets: [
      {
        name: "Line 1",
        color: "#ff0",
        data: [1, 2, 3, 4],
      },
      {
        name: "Line 2",
        color: "#f0f",
        data: [8, 4, 2, 1],
      },
      {
        name: "Line 3",
        color: "#0ff",
        data: [1, 4, 1, 4],
      },
    ],
  },
});
```
